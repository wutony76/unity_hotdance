Replay.StartRecording

public static void StartRecording (Camera recordingCamera, Configuration configuration, VideoCallback replayCallback, IAudioSource audioSource);

Description

Start recording a replay with optional audio. 

public class Marshmellows : MonoBehaviour {

    // Called when the level starts
    void Start () {
        // Start recording a replay from the game camera
        Replay.StartRecording(Camera.main, Configuration.Default, replayPath => DoStuff(replayPath));
        // Wait a while
        yield return new WaitForSeconds(10f);
        // Stop recording // The provided callback will be invoked with the path to the recorded video
        Replay.StopRecording();
    }
}
